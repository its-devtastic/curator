# contentTypes

`ContentTypeConfig[]`

A list of content type definitions.

### Types

```ts
interface ContentTypeConfig {
  /** The apiID is how Strapi identifies the content type. */
  apiID: string;
  fields: FieldDefinition[];
  /** Name and description can be translation keys or literal strings */
  name?: string;
  description?: string;
  icon?: React.ReactNode;
  getEntityUrl?(entity: Entity): string;
}

export interface FieldDefinition {
  /** The path to the key of the field (e.g. "username") */
  path: string;
  /** Label, description and hint can be translation keys or literal strings */
  label?: string;
  description?: string;
  hint?: string;
  input?: string;
  inputProps?: Record<string, any>;
  /** Only for relation and enum fields */
  renderItem?(item: any, utils: { t: TFunction }): React.ReactNode;
}
```
